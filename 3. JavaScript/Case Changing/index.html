Strings: Finding Segments** Example 1: Find the Position of a Word using `indexOf()` let sentence = "Learning JavaScript is fun"; let position = sentence.indexOf("JavaScript"); console.log(position); // Output: 9 Example 2: Find the Position of a Word that Doesn't Exist let sentence = "Learning JavaScript is fun"; let position = sentence.indexOf("Python"); console.log(position); // Output: -1 (meaning "Python" is not found) Example 3: Find the Last Occurrence using `lastIndexOf()` let sentence = "JavaScript is great, and JavaScript is fun"; let lastPosition = sentence.lastIndexOf("JavaScript"); console.log(lastPosition); // Output: 27 Example 4: Find Position from a Specific Starting Point let sentence = "JavaScript is great, and JavaScript is fun"; let position = sentence.indexOf("JavaScript", 10); // Start searching after index 10 console.log(position); // Output: 27 Example 5: Case-Sensitive Search with `indexOf()` let sentence = "Learning JavaScript is Fun"; let position = sentence.indexOf("javascript"); // Case-sensitive, won't match console.log(position); // Output: -1 4. Strings: Finding a Character at a Location Example 1: Finding a Character Using `charAt()` let text = "JavaScript"; console.log(text.charAt(4)); // Output: "S" Example 2: Accessing Character Directly let text = "JavaScript"; console.log(text[4]); // Output: "S" (same as charAt) Example 3: Finding Character at the Last Position let text = "JavaScript"; console.log(text.charAt(text.length - 1)); // Output: "t" Example 4: Using `charCodeAt()` to Get Unicode Value of a Character let text = "A"; console.log(text.charCodeAt(0)); // Output: 65 (Unicode value of 'A') Example 5: Check if a Character Exists at a Specific Index let text = "Hello World"; if (text[15]) { console.log("Character exists at index 15."); } else { console.log("No character at index 15."); // Output: "No character at index 15." } Strings: Replacing Characters Example 1: Simple String Replacement using `replace()` let sentence = "I like apples"; let newSentence = sentence.replace("apples", "bananas"); console.log(newSentence); // Output: "I like bananas" Example 2: Replacing Only the First Occurrence let sentence = "Apples are apples"; let newSentence = sentence.replace("apples", "bananas"); console.log(newSentence); // Output: "Apples are bananas" Example 3: Using Regular Expressions for Case-Insensitive Replacement let sentence = "Apples are Apples"; let newSentence = sentence.replace(/apples/i, "bananas"); // "i" for case-insensitive console.log(newSentence); // Output: "bananas are Apples" Example 4: Replacing All Occurrences Using Regex let sentence = "Apples are apples"; let newSentence = sentence.replace(/apples/g, "bananas"); // "g" for global replacement console.log(newSentence); // Output: "bananas are bananas" Example 5: Replacing Multiple Words Using `replace()` let sentence = "I love cats and dogs"; let newSentence = sentence.replace("cats", "birds").replace("dogs", "fish"); console.log(newSentence); // Output: "I love birds and fish"
